\begin{algorithm}[!htbp]
\caption{Two-stage greedy subcover with ratio $\eta$}
\label{alg:eta_greedy_subcover}
\begin{algorithmic}[1]
\Require Distance matrix $D\in\mathbb{R}^{M\times|\mathcal{S}|}$; threshold ratio $\eta\in(0,1)$
\State $\tau \gets\max_m \min_{s\in\mathcal{S}}D_{m,s}$
\State $\tau' \gets \eta\tau$
\State $A_{\tau'}\gets\ind\{D\leq\tau'\}$
\State $A_\tau\gets\ind\{D\leq\tau\}$
\State $(\mathcal{S}_{\tau'},U_{\tau'}) \gets \Call{GreedySubcover}{A_{\tau'}}$
\State $A_\tau^{\mathrm{rem}}\gets A_\tau$ restricted to rows $U_{\tau'}$ and columns $\mathcal{S}\setminus\mathcal{S}_{\tau'}$
\State $(\mathcal{S}_\tau,U_\tau) \gets \Call{GreedySubcover}{A_\tau^{\mathrm{rem}}}$
\State $\mathcal{S}_\eta \gets \mathcal{S}_{\tau'}\cup\mathcal{S}_\tau$
\State \Return $\mathcal{S}_\eta$
\end{algorithmic}
\end{algorithm}
